const getCategoryPath = (categoryName: string, subCategoryName?: string, subSubCategoryName?: string) => {
  let path = categoryName;
  if (subCategoryName) path += ` > ${subCategoryName}`;
  if (subSubCategoryName) path += ` > ${subSubCategoryName}`;
  return path;
};

const findCategoryByName = (name: string) => {
    if (!name) return null;
    const pathParts = name.split(' > ');
    let currentLevel: any = categoriesConfig;
    let foundCategory = null;

    for (const part of pathParts) {
        const category = currentLevel.find((cat: any) => cat.name === part);
        if (category) {
            foundCategory = category;
            currentLevel = category.subcategories || [];
        } else {
            return null; 
        }
    }
    return foundCategory;
};

export { categoriesConfig, getCategoryPath, findCategoryByName };
